package api

// Book implements OpenAPI element at #/components/schemas/Book
// A financial book, holding accounts, categories and transactions.
type Book struct {
	ID      BookId `json:"id"`
	Name    string `json:"name"`
	OwnerID UserId `json:"owner_id"`
}

// BookId implements OpenAPI element at #/components/schemas/BookId
// Unique identifier of a financial book in the system.
type BookId int

// BookList implements OpenAPI element at #/components/schemas/BookList
// A list of books.
type BookList struct {
	Items []Book `json:"items"`
}

// UserId implements OpenAPI element at #/components/schemas/UserId
// Unique identifier of a user in the system.
type UserId int
